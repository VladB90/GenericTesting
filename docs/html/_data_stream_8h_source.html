175 template<class T>
176 void DataStream<T>::AddData(std::vector<T>&& data, float firstSampleTime, float lastSampleTime)
177 {
178 std::unique_lock<std::mutex> lk(dataLock);
179
180 // If sample time given is before the last sample time added, toss out request to add data
181 if (firstSampleTime < lastSampleTimeInserted || lastSampleTime < lastSampleTimeInserted)
182 {
183 return;
184 }
185
186 // Calculate number of frames to add to start of data due to sample time difference
187 float timeToLastSample = firstSampleTime - lastSampleTimeInserted;
188 if (timeToLastSample > 1.0f / dataFrameRate)
189 {
190 int framesToPad = static_cast<int>(std::floor((timeToLastSample * dataFrameRate) + dataFrameRoundTolerance));
191
192 for (size_t i = 0; i < framesToPad; i++)
193 {
194 data.insert(data.begin(), T());
195 }
196 }
197
198 lk.unlock();
199 storedData.Push(data);
200
201 for (size_t i = 0; i < storedConditions.size(); i++)
202 {
203 BaseCondition listener = storedConditions[i];
204
205 if (listener.HasValidConditions(nextFrameIndex, storedData.GetBufferCapacity()))
206 {
207 std::lock_guard<std::mutex> lock(*listener.dataLock);
208 listener.conditionVar->notify_all();
209 }
210 }
211
212
213 lastSampleTimeInserted = lastSampleTime;
214 nextFrameIndex += data.size();
215 }

219 {
220 std::unique_lock<std::mutex> lk(dataLock);
221
222 // NOTE: This only accounts for if data is only removed from the circular buffer by being overwritten. If data is manually removed, then this would be incorrect.
223 int oldestFrameInBuffer = nextFrameIndex - storedData.GetBufferCapacity();
224 if (frameNumber <= oldestFrameInBuffer || frameNumber > nextFrameIndex)
225 {
226 return T();
227 }
228
229 int startIndex = frameNumber % storedData.GetBufferCapacity();
230
231
232 std::vector<T> copyData;
233 copyData.reserve(1);
234
235 storedData.CopyToBuffer(frameNumber, 1, copyData);
236
237 T data = copyData[0];
238 return data;

293 {
294 std::unique_lock<std::mutex> lk(dataLock);
295
296 if (nextFrameIndex < 0)
297 {
298 // No data in stream, return empty;
299 return std::vector<T>();
300 }
301
302 int bufferCalc = nextFrameIndex - storedData.GetBufferCapacity();
303
304 int oldestFrameInBuffer;
